function Card({
  icon,
  title,
  points,
}: {
  icon: React.ReactNode;
  title: string;
  points: string[];
}) {
  return (
    <motion.div
      whileHover={{ y: -10 }}
      className="bg-white/5 border border-white/10 backdrop-blur-xl rounded-3xl p-10 hover:border-yellow-500/40 transition-all"
    >
      <div className="text-yellow-400 mb-6">
        {icon}
      </div>

      <h3 className="text-3xl font-bold mb-6">
        {title}
      </h3>

      <ul className="space-y-4 text-gray-300">
        {points.map((point, index) => (
          <li key={index}>
            • {point}
          </li>
        ))}
      </ul>
    </motion.div>
  );
}
